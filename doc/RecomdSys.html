def ecludSim(inA,inB):
    return 1.0 / (1.0 + la.norm(inA - inB))

def pearsSim(inA,inB):
    if len(inA) < 3 : return 1.0
    return 0.5 + 0.5 * corrcoef(inA, inB, rowvar = 0)[0][1]

def distCosin(v1,v2): #------------------------------------------>> standEst method
        D_norm = la.norm(v1)
        x_norm = la.norm(v2)
        cosines = dot(v1,v2)/(D_norm * x_norm) 
        distances = 1 - cosines
        return distances

def KMEANS(data,n):
    kmeans = KMeans(n_clusters=n,max_iter=500)
    kmeans.fit(data)
    clusters = kmeans.predict(data)
    dataout = pd.DataFrame(data)
    dataout['cluster'] = clusters
    datamtx = array(dataout)
    return datamtx
